    Create packages/proxy/src/proxy.ts with createProxyMiddleware setup:

    ```typescript
    import { createProxyMiddleware } from 'http-proxy-middleware';
    import { providerService } from './services/provider';
    import { keychainService } from './services/keychain'; // imported, not yet implemented
    import type { RouteResolution } from './types';

    export function createProxyHandler() {
      return createProxyMiddleware({
        // Per D-02: port 3456 default, configurable
        target: 'http://localhost:8000', // fallback only
        changeOrigin: true,
        pathRewrite: { '/v1': '/v1' }, // preserve path
        selfHandleResponse: false,    // SSE passthrough (PROX-03)
        // Dynamic router resolves target at request time (PROX-02)
        router: async (req) => {
          // Parse model from request body — http-proxy-middleware v3 supports body parsing
          const body = (req as any).body || {};
          const modelName: string = body.model || 'claude-opus-4-20250514';

          const resolution: RouteResolution | null = providerService.resolveModelRoute(modelName);
          if (!resolution) {
            throw new Error(`No route for model: ${modelName}`);
          }

          // Store for onProxyReq hook — API key fetched there
          (req as any).routeResolution = resolution;

          return resolution.provider.baseUrl;
        },
        onProxyReq: (proxyReq, req) => {
          const resolution: RouteResolution | null = (req as any).routeResolution;
          if (!resolution) return;

          // Inject Bearer token with API key from Keychain
          // keychainService.getKey() is async — use request extension to store resolved key
          // For now, use a placeholder header; Plan 02 wires the full async flow
          // NOTE: This is a synchronous hook — async keychain fetch must be pre-resolved
          // Solution: resolver async router sets Authorization header before proxyReq fires
        },
        onProxyRes: (proxyRes, req, res) => {
          // Stream passthrough — no buffering (PROX-03)
          // http-proxy-middleware selfHandleResponse: false handles this
        },
        onError: (err, req, res) => {
          console.error('[Proxy Error]', err.message);
          if ((res as any).writable) {
            (res as any).writeHead(502, { 'Content-Type': 'application/json' });
            (res as any).end(JSON.stringify({ error: 'Provider routing failed', detail: err.message }));
          }
        },
      });
    }
    ```

    6. `startServer(port = 3456): Promise<void>` — exports for testing

    Key insight (from RESEARCH.md PITFALL-5): `selfHandleResponse: false` ensures SSE passthrough. The async router function (PROX-02) resolves the target per-request. API key injection happens in the onProxyReq hook — plan 02 wires the async keychain fetch properly.

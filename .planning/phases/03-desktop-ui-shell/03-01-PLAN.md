    Write src-tauri/src/lib.rs with three Tauri commands for proxy lifecycle:

    ```rust
    use tauri::Manager;
    use std::process::Command;
    use std::sync::Mutex;

    // Global state to track the proxy child process PID
    struct ProxyState {
        pid: Mutex<Option<u32>>,
    }

    #[tauri::command]
    async fn start_proxy(app: tauri::AppHandle) -> Result<serde_json::Value, String> {
        let state = app.state::<ProxyState>();
        let mut pid_lock = state.pid.lock().map_err(|e| e.to_string())?;

        if let Some(existing_pid) = *pid_lock {
            // Check if process is still running
            // If running, return already-running response
            return Ok(serde_json::json!({
                "success": false,
                "message": "Proxy is already running",
                "pid": existing_pid
            }));
        }

        // Spawn the proxy: node packages/proxy/dist/index.js
        // Use the app's resource directory or resolve relative to app path
        let mut cmd = Command::new("node");
        cmd.arg("packages/proxy/dist/index.js");
        cmd.env("NODE_ENV", "production");

        let child = cmd.spawn().map_err(|e| format!("Failed to start proxy: {}", e))?;
        let pid = child.id();
        *pid_lock = Some(pid);

        // Detach so it keeps running
        // Note: In production, manage the process lifecycle properly
        Ok(serde_json::json!({
            "success": true,
            "port": 3456,
            "pid": pid
        }))
    }

    #[tauri::command]
    async fn stop_proxy(app: tauri::AppHandle) -> Result<serde_json::Value, String> {
        let state = app.state::<ProxyState>();
        let mut pid_lock = state.pid.lock().map_err(|e| e.to_string())?;

        if let Some(pid) = pid_lock.take() {
            // Kill the process by PID
            #[cfg(unix)]
            {
                use std::process::Command;
                let _ = Command::new("kill").arg(pid.to_string()).output();
            }
            #[cfg(windows)]
            {
                use std::process::Command;
                let _ = Command::new("taskkill")
                    .args(["/F", "/PID", &pid.to_string()])
                    .output();
            }
            Ok(serde_json::json!({
                "success": true,
                "message": "Proxy stopped"
            }))
        } else {
            Ok(serde_json::json!({
                "success": false,
                "message": "Proxy is not running"
            }))
        }
    }

    #[tauri::command]
    async fn get_proxy_status() -> Result<serde_json::Value, String> {
        // Check health endpoint at localhost:3456/health
        let client = reqwest::blocking::Client::builder()
            .timeout(std::time::Duration::from_secs(2))
            .build()
            .map_err(|e| e.to_string())?;

        match client.get("http://localhost:3456/health").send() {
            Ok(resp) if resp.status().is_success() => {
                let body: serde_json::Value = resp.json().map_err(|e| e.to_string())?;
                Ok(serde_json::json!({
                    "running": true,
                    "status": body.get("status").and_then(|s| s.as_str()).unwrap_or("unknown"),
                    "port": body.get("port").and_then(|p| p.as_u64()).unwrap_or(3456),
                    "version": body.get("version").and_then(|v| v.as_str()).unwrap_or("unknown")
                }))
            }
            _ => Ok(serde_json::json!({
                "running": false,
                "status": "stopped",
                "port": 3456,
                "version": null
            }))
        }
    }

    #[cfg_attr(mobile, tauri::mobile_entry_point)]
    pub fn run() {
        tauri::Builder::default()
            .plugin(tauri_plugin_shell::init())
            .manage(ProxyState {
                pid: Mutex::new(None),
            })
            .invoke_handler(tauri::generate_handler![
                start_proxy,
                stop_proxy,
                get_proxy_status
            ])
            .run(tauri::generate_context!())
            .expect("error while running Tauri application");
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Tauri → Express child process | Untrusted: Express runs as separate Node.js process with its own privileges |
| Frontend → Proxy (localhost:3456) | Trusted: localhost-only, no network exposure, but any local process can access it |
| Rust commands → filesystem | Trusted: Tauri sandboxed, but spawns external processes |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | Express sidecar | mitigate | Bind Express to 127.0.0.1 only (not 0.0.0.0), validate host header in middleware |
| T-03-02 | Tampering | Proxy child process | mitigate | Rust command verifies PID ownership before kill, uses process group to prevent orphan processes |
| T-03-03 | Information Disclosure | Health endpoint | accept | Health endpoint returns only status/port/version — no sensitive data, localhost-only |
| T-03-04 | Elevation of Privilege | Shell spawn permission | mitigate | Tauri capability scoped to "main" window only, no filesystem write permissions granted |
</threat_model>
